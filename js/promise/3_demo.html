Promise 文件读取演示

这个页面演示了3.js中的Promise概念，模拟了文件读取操作和错误处理。

const p = new Promise((resolve, reject) => {
  console.log(3); // 同步，立即执行
  
  // 模拟文件读取操作
  setTimeout(() => {
    const success = Math.random() > 0.5; // 50%概率成功
    if (success) {
      resolve("文件内容: hello world");
    } else {
      reject(new Error("文件不存在: b.txt"));
    }
  }, 1000);
});

p.then((data) => {
  console.log(data, '读取文件成功');
}).catch((err) => {
  console.log(err, '读取文件失败');
});

console.log(2); // 同步
