941. 有效的山脉数组

给定一个整数数组 arr，如果它是有效的山脉数组就返回 true，否则返回 false。


/**
 * @param {number[]} arr
 * @return {boolean}
 */
//  双阶段扫描法，核心思想是：
// 一个有效的“山脉数组”必须严格先递增、再严格递减，且峰值不能在首尾。
var validMountainArray = function(arr) {
    const N = arr.length;
    let i = 0;

    // 从左往右找“上坡”（严格递增）
    while (i + 1 < N && arr[i] < arr[i + 1]) {
        i++;
    }

    // 检查山顶是否合法：
    if (i === 0 || i === N - 1) {
        return false;
    }

    // 从山顶开始找“下坡”（严格递减）
    while (i + 1 < N && arr[i] > arr[i + 1]) {
        i++;
    }
    //最终判断是否走完整个数组,返回true/false
    return i === N - 1;
};
