/\*\*

- 将给定的 n×n 矩阵原地顺时针旋转 90 度。
- 注意：虽然函数名是 "rotate"，但实际实现是通过辅助矩阵完成的，并非严格“原地”操作。
-
- @param {number[][]} matrix - 输入的二维数组（方阵）
  \*/
  var rotate = function(matrix) {
  const n = matrix.length; // 获取矩阵的维度 n（因为是 n×n 方阵）

      // 创建一个新的 n×n 矩阵，初始化为 0，用于存放旋转后的结果
      const matrix_new = new Array(n).fill(0).map(() => new Array(n).fill(0));

      // 遍历原矩阵的每一个元素
      for (let i = 0; i < n; i++) {
          for (let j = 0; j < n; j++) {
              // 核心旋转逻辑：
              // 原矩阵中位置 (i, j) 的元素，
              // 在顺时针旋转 90 度后，会移动到新矩阵的 (j, n - i - 1) 位置。
              matrix_new[j][n - i - 1] = matrix[i][j];
          }
      }

      // 将新矩阵的内容复制回原矩阵，以满足“修改原矩阵”的要求
      for (let i = 0; i < n; i++) {
          for (let j = 0; j < n; j++) {
              matrix[i][j] = matrix_new[i][j];
          }
      }

  };
