var moveZeroes = function(nums) {
let zeroBoundary = 0; // 指向下一个非零元素应该放的位置

    for (let cur = 0; cur < nums.length; cur++) {
        if (nums[cur] !== 0) {
            // 将非零元素放到 zeroBoundary 的位置
            nums[zeroBoundary] = nums[cur];
            zeroBoundary++;
        }
    }

    // 将从 zeroBoundary 到末尾的所有元素设为 0
    while (zeroBoundary < nums.length) {
        nums[zeroBoundary] = 0;
        zeroBoundary++;
    }

};
